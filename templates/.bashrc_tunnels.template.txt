# Target jump server (replace with your actual user@ip)
SSH_DEST="user@JUMP_SERVER_IP"

# Function to create an SSH tunnel
start_ssh_tunnel() {
  local local_port="$1"
  local target_host="$2"
  local remote_port="$3"
  local dest="$4"

  echo "ðŸ”— Starting tunnel: localhost:$local_port -> $target_host:$remote_port via $dest"
  ssh -f -N -L "${local_port}:${target_host}:${remote_port}" "$dest"
}
